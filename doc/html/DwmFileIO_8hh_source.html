512 template <typename... Ts>
513 static size_t Read(FILE *fp, std::variant<Ts...> & v)
514 {
515 size_t rc = 0;
516 uint64_t index = 0;
517 if (Read(fp, index)) {
518 if (index < std::variant_size_v<std::variant<Ts...>>) {
519 v = VariantFromIndex<Ts...>(index);
520 std::visit([&] (auto && arg) { rc = Read(fp, arg); }, v);
521 }
522 }
523 return rc;
524 }

530 static size_t Write(FILE *fp, const std::variant<Ts...> & v)
531 {
532 size_t rc = 0;
533 uint64_t index = v.index();
534 if (Write(fp, index)) {
535 std::visit([&] (const auto & arg) { rc = Write(fp, arg); }, v);
536 }
537 return rc;

660 template <typename _inputIteratorT>
661 static size_t Write(FILE *file, _inputIteratorT f, _inputIteratorT l)
662 {
663 size_t rc = 0;
664 if (file) {
665 for ( ; f != l; ++f) {
666 if (Write(file, *f) <= 0) {
667 break;
668 }
669 }
670 if (f == l)
671 rc = 1;
672 }
673 return(rc);
674 }

680 template <typename _containerT>
681 static size_t ContainerRead(FILE *f, _containerT & c)
682 {
683 size_t rc = 0;
684 if (! c.empty())
685 c.clear();
686 if (f) {
687 uint64_t numEntries;
688 if (Read(f, numEntries)) {
689 uint64_t i = 0;
690 for ( ; i < numEntries; ++i) {
691 typename _containerT::value_type val;
692 if (Read(f, val) > 0)
693 c.insert(c.end(), std::move(val));
694 else
695 break;
696 }
697 if (i == numEntries)
698 rc = 1;
699 }
700 }
701 return(rc);
702 }
703
704 //------------------------------------------------------------------------
707 //------------------------------------------------------------------------
708 template <typename _containerT>
709 static size_t ContainerWrite(FILE *f, const _containerT & c)
710 {
711 size_t rc = 0;
712 if (f) {
713 uint64_t numEntries = c.size();
714 if (Write(f, numEntries)) {
715 if (numEntries) {
716 rc = Write<typename _containerT::const_iterator>(f,
717 c.begin(), c.end());
718 }
719 else {
720 rc = 1;
721 }
722 }
723 }
724 return(rc);
725 }

732 template <typename _containerT>
733 static size_t PairAssocContRead(FILE *f, _containerT & m)
734 {
735 ssize_t rc = 0;
736 if (! m.empty())
737 m.clear();
738 if (f) {
739 uint64_t numEntries;
740 if (Read(f, numEntries)) {
741 uint64_t i = 0;
742 for ( ; i < numEntries; ++i) {
743 typename _containerT::key_type key;
744 if (Read(f, key) > 0) {
745 typename _containerT::mapped_type val;
746 if (Read(f, val) > 0) {
747 m.insert(typename _containerT::value_type(std::move(key),
748 std::move(val)));
749 }
750 else {
751 break;
752 }
753 }
754 else {
755 break;
756 }
757 }
758 if (i == numEntries)
759 rc = 1;
760 }
761 }
762 return(rc);
763 }
