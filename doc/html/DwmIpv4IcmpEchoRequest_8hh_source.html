118 template <typename T>
119 bool SetData(const T & data)
120 {
121 bool rc = false;
122 uint32_t dataLen = IOUtils::StreamedLength(data);
123 Reallocate(dataLen + 8);
124 std::ostringstream os;
125 if (StreamIO::Write(os, data)) {
126 memcpy((char *)_data.second + 8, os.str().c_str(), dataLen);
127 rc = true;
128 }
129 return rc;
130 }
131
132 //------------------------------------------------------------------------
137 //------------------------------------------------------------------------
138 template <typename T>
139 bool GetData(T & data) const
140 {
141 bool rc = false;
142 if (_data.first >= (8 + IOUtils::StreamedLength(data))) {
143 std::istringstream is(std::string((char *)_data.second + 8,
144 _data.first - 8));
145 rc = (StreamIO::Read(is, data) ? true : false);
146 }
147 return(rc);
148 }
