56 template <typename... Ts>
57 std::variant<Ts...> VariantFromIndex(size_t index)
58 {
59 assert(index < sizeof...(Ts));
60 static constexpr auto tbl =
61 std::array{ +[](){ return std::variant<Ts...>{Ts{ }}; }... };
62 return tbl[index]();
