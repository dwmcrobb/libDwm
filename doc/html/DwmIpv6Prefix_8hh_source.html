293 template <typename S>
294 requires IsSupportedASIOSocket<S>
295 bool ASIO_Read(S & s, boost::system::error_code & ec)
296 {
297 using boost::asio::read, boost::asio::buffer;
298 bool rc = false;
299 if ((read(s, buffer(&_length, sizeof(_length)), ec) == sizeof(_length))
300 && (! ec)) {
301 size_t len = (_length + 7) >> 3;
302 if ((read(s, buffer(_addr.s6_addr, len), ec) == sizeof(_length))
303 && (! ec)) {
304 rc = true;
305 }
306 }
307 return rc;
308 }
