157 inline uint8_t MaskLength(uint8_t maskLength)
158 {
159 if (maskLength < _data[4]) {
160 if (maskLength == 0) {
161 *(uint32_t *)_data = 0;
162 }
163 else {
164 // need to zero some bits
165 uint32_t mask = htonl(0xffffffff << (32 - maskLength));
166 *(uint32_t *)_data &= mask;
167 }
168 _data[4] = maskLength;
169 }
170 else if (maskLength < 33) {
171 _data[4] = maskLength;
172 }
173 return(_data[4]);
174 }
