294 //------------------------------------------------------------------------
295 bool Remove(std::unique_lock<std::shared_mutex> & lck,
296 const Ipv6Prefix & pfx)
297 {
298 assert((lck.mutex() == &_mtx) && lck.owns_lock());
299 bool rc = false;
300 auto it = _maps.first.find(pfx);
301 if (it != _maps.first.end()) {
302 _maps.first.erase(it);
303 auto lit = _maps.second.find(pfx.MaskLength());
304 lit->second--;
305 if (0 == lit->second) {
306 _maps.second.erase(lit);
307 }
308 rc = true;
309 }
310 return rc;
311 }

586 bool FindNoLock(const Ipv6Prefix & pfx, T & value) const
587 {
588 bool rc = false;
589 auto it = _maps.first.find(pfx);
590 if (it != _maps.first.end()) {
591 value = it->second;
592 rc = true;
593 }
594 return rc;
595 }
596
597 //------------------------------------------------------------------------
599 //------------------------------------------------------------------------
600 bool FindLongestNoLock(const Ipv6Address & addr,
601 std::pair<Ipv6Prefix,T> & value) const
602 {
603 bool rc = false;
604 value.first.Set(addr, 128);
605 for (auto lit = _maps.second.rbegin();
606 lit != _maps.second.rend(); ++lit) {
607 value.first.MaskLength(lit->first);
608 auto it = _maps.first.find(value.first);
609 if (it != _maps.first.end()) {
610 value.second = it->second;
611 rc = true;
612 break;
613 }
614 }
615 return rc;
616 }
617
618 //------------------------------------------------------------------------
620 //------------------------------------------------------------------------
621 bool FindMatchesNoLock(const Ipv6Address & addr,
622 std::vector<std::pair<Ipv6Prefix,T>> & values) const
623 {
624 values.clear();
625 std::pair<Ipv6Prefix,T> value;
626 for (auto lit = _maps.second.rbegin();
627 lit != _maps.second.rend(); ++lit) {
628 Ipv6Prefix pfx(addr, lit->first);
629 auto it = _maps.first.find(pfx);
630 if (it != _maps.first.end()) {
631 value.first = pfx;
632 value.second = it->second;
633 values.push_back(value);
634 }
635 }
636 return (! values.empty());
637 }
