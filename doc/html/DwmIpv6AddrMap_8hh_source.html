65 {

67 {

155 bool Remove(std::unique_lock<std::shared_mutex> & lck,
156 const Ipv6Address & addr)
157 {
158 bool rc = false;
159 assert((lck.mutex() == &_mtx) && lck.owns_lock());
160 auto it = _map.find(addr);
161 if (it != _map.end()) {
162 _map.erase(it);
163 rc = true;
164 }
165 return rc;
166 }

418 //------------------------------------------------------------------------
419 bool FindNoLock(const Ipv6Address & addr, T & value) const
420 {
421 auto iter = _map.find(addr);
422 if (iter != _map.end()) {
423 value = iter->second;
424 return true;
425 }
426 return false;
