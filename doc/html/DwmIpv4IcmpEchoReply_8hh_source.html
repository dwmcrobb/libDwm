114 template <typename T>
115 bool SetData(const T & data)
116 {
117 bool rc = false;
118 uint32_t dataLen = IOUtils::StreamedLength(data);
119 Reallocate(dataLen + 8);
120 std::ostringstream os;
121 if (StreamIO::Write(os, data)) {
122 memcpy((char *)_data.second + 8, os.str().c_str(), dataLen);
123 rc = true;
124 }
125 return rc;
126 }
127
128 //------------------------------------------------------------------------
133 //------------------------------------------------------------------------
134 template <typename T>
135 bool GetData(T & data) const
136 {
137 bool rc = false;
138 if (_data.first >= (8 + IOUtils::StreamedLength(data))) {
139 std::istringstream is(std::string((char *)_data.second + 8,
140 _data.first - 8));
141 if (StreamIO::Read(is, data)) {
142 rc = true;
143 }
144 }
145 return(rc);
146 }
