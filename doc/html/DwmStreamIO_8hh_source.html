581 template <typename... Ts>
582 static std::istream & Read(std::istream & is,
583 std::variant<Ts...> & v)
584 {
585 uint64_t index = 0;
586 if (Read(is, index)) {
587 if (index < std::variant_size_v<std::variant<Ts...>>) {
588 v = VariantFromIndex<Ts...>(index);
589 std::visit([&] (auto && arg) { Read(is, arg); }, v);
590 }
591 else {
592 is.setstate(std::ios_base::failbit);
593 }
594 }
595 return is;
596 }

601 template <typename... Ts>
602 static std::ostream & Write(std::ostream & os,
603 const std::variant<Ts...> & v)
604 {
605 uint64_t index = v.index();
606 if (Write(os, index)) {
607 std::visit([&os] (const auto & arg) { Write(os, arg); }, v);
608 }
609 return os;
610 }

678 template <typename _containerT>
679 static std::istream & ContainerRead(std::istream & is,
680 _containerT & c)
681 {
682 if (! c.empty())
683 c.clear();
684 if (is) {
685 uint64_t numEntries;
686 if (Read(is, numEntries)) {
687 for (uint64_t i = 0; i < numEntries; ++i) {
688 typename _containerT::value_type val;
689 if (! Read(is, val))
690 break;
691 c.insert(c.end(), std::move(val));
692 }
693 }
694 }
695 return(is);
696 }

701 //------------------------------------------------------------------------
702 template <typename _containerT>
703 static std::ostream & ContainerWrite(std::ostream & os,
704 const _containerT & c)
705 {
706 if (os) {
707 uint64_t numEntries = c.size();
708 if (Write(os, numEntries)) {
709 if (numEntries) {
710 Write<typename _containerT::const_iterator>(os,
711 c.begin(),
712 c.end());
713 }
714 }
715 }
716 return(os);
717 }

724 template <typename _containerT>
725 static std::istream &
726 PairAssocContRead(std::istream & is, _containerT & m)
727 {
728 if (! m.empty())
729 m.clear();
730 if (is) {
731 uint64_t numEntries;
732 if (Read(is, numEntries)) {
733 for (uint64_t i = 0; i < numEntries; ++i) {
734 typename _containerT::key_type key;
735 if (Read(is, key)) {
736 typename _containerT::mapped_type val;
737 if (Read(is, val))
738 m.insert(typename _containerT::value_type(std::move(key), std::move(val)));
739 else
740 break;
741 }
742 else
743 break;
744 }
745 }
746 }
747 return(is);
748 }
