71 template <typename T>
72 std::string Bold(const T & t) const
73 {
74 std::ostringstream os;
75 if (_on) {
76 os.imbue(std::locale(""));
77 os << _startBold;
78 }
79 os << t;
80 if (_on)
81 os << _endBold;
82 return(os.str());
83 }
