267 template <typename _firstT, typename _secondT>
268 static ssize_t Read(int fd, std::pair<_firstT, _secondT> & p)
269 {
270 ssize_t rc = -1;
271 if (fd >= 0) {
272 ssize_t bytesRead = Read(fd, p.first);
273 if (bytesRead > 0) {
274 rc = bytesRead;
275 bytesRead = Read(fd, p.second);
276 if (bytesRead > 0)
277 rc += bytesRead;
278 else
279 rc = -1;
280 }
281 }
282 return(rc);

668 template <typename... Ts>
669 static ssize_t Read(int fd, std::variant<Ts...> & v)
670 {
671 ssize_t rc = -1;
672 uint64_t index = 0;
673 ssize_t bytesRead = Read(fd, index);
674 if (bytesRead > 0) {
675 rc = bytesRead;
676 if (index < std::variant_size_v<std::variant<Ts...>>) {
677 v = VariantFromIndex<Ts...>(index);
678 std::visit([&] (auto && arg) { bytesRead = Read(fd, arg); }, v);
679 if (bytesRead >= 0) {
680 rc += bytesRead;
681 }
682 else {
683 rc = -1;
684 }
685 }
686 else {
687 rc = -1;
688 }
689 }
690 return rc;
691 }

803 template <typename _containerT>
804 static ssize_t ContainerRead(int fd, _containerT & c)
805 {
806 if (! c.empty())
807 c.clear();
808 ssize_t rc = -1;
809 if (fd >= 0) {
810 uint64_t numEntries;
811 ssize_t bytesRead = Read(fd, numEntries);
812 if (bytesRead == sizeof(numEntries)) {
813 rc = bytesRead;
814 for (uint64_t i = 0; i < numEntries; ++i) {
815 typename _containerT::value_type val;
816 bytesRead = Read(fd, val);
817 if (bytesRead > 0) {
818 rc += bytesRead;
819 c.insert(c.end(), std::move(val));
820 }
821 else {
822 rc = -1;
823 break;
824 }
825 }
826 }
827 }
828 return(rc);
829 }

836 template <typename _containerT>
837 static ssize_t ContainerWrite(int fd, const _containerT & c)
838 {
839 ssize_t rc = -1;
840 if (fd >= 0) {
841 uint64_t numEntries = c.size();
842 uint64_t bytesWritten = Write(fd, numEntries);
843 if (bytesWritten == sizeof(numEntries)) {
844 rc = bytesWritten;
845 if (numEntries) {
846 bytesWritten =
847 Write<typename _containerT::const_iterator>(fd,
848 c.begin(),
849 c.end());
850 if (bytesWritten > 0)
851 rc += bytesWritten;
852 else
853 rc = -1;
854 }
855 }
856 }
857 return(rc);

865 template <typename _containerT>
866 static ssize_t PairAssocContRead(int fd, _containerT & m)
867 {
868 ssize_t rc = -1;
869 if (! m.empty())
870 m.clear();
871 if (fd >= 0) {
872 uint64_t numEntries;
873 ssize_t bytesRead = Read(fd, numEntries);
874 if (bytesRead == sizeof(numEntries)) {
875 rc = bytesRead;
876 for (uint64_t i = 0; i < numEntries; ++i) {
877 typename _containerT::key_type key;
878 bytesRead = Read(fd, key);
879 if (bytesRead > 0) {
880 rc += bytesRead;
881 typename _containerT::mapped_type val;
882 bytesRead = Read(fd, val);
883 if (bytesRead > 0) {
884 rc += bytesRead;
885 m.insert(typename _containerT::value_type(std::move(key),
886 std::move(val)));
887 }
888 else {
889 rc = -1;
890 break;
891 }
892 }
893 else {
894 rc = -1;
895 break;
896 }
897 }
898 }
899 }
900 return(rc);
901 }
