129 template <typename ...Args>
130 static bool FmtLog(const std::string & filename, int lineno,
131 const std::string & function, int priority,
132 std::format_string<Args...> fm,
133 Args&&... args)
134 {
135 // The reason I put the formatted string into the format string
136 // of the downstream Log() call instead of separately using a
137 // "%s": I want to be able to still use the convenient "%m" in
138 // a log message, and it must be in the format string for that
139 // to happen downstream. This is likely to change at a later date.
140 return Log(filename, lineno, function, priority,
141 std::format(fm,std::forward<Args>(args)...).c_str());
142 }
