124 template <typename T>
125 bool GetPayload(T & payload) const
126 {
127 bool rc = false;
128 if (_payload.first) {
129 std::istringstream is(std::string((const char *)_payload.second,
130 _payload.first));
131 rc = StreamIO::Read(is, payload);
132 }
133 return(rc);
134 }
135
136 //------------------------------------------------------------------------
140 //------------------------------------------------------------------------
141 template <typename T>
142 bool SetPayload(const T & payload)
143 {
144 bool rc = false;
145 Free();
146
147 uint16_t maxPayloadLen = 65535 - (20 + 8);
148 uint32_t payloadLen = IOUtils::StreamedLength(payload);
149 if (payloadLen <= maxPayloadLen) {
150 std::ostringstream os;
151 if (StreamIO::Write(os, payload)) {
152 _payload.first = payloadLen;
153 _payload.second = (uint8_t *)calloc(1, _payload.first);
154 assert(_payload.second);
155 memcpy(_payload.second, os.str().c_str(), _payload.first);
156 rc = true;
157 }
158 }
159 return(rc);
160 }
