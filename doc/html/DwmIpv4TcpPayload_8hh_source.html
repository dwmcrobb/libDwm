129 template <typename T>
130 bool GetPayload(T & payload) const
131 {
132 bool rc = false;
133 if (_payload.first) {
134 std::istringstream is(std::string((const char *)_payload.second,
135 _payload.first));
136 rc = StreamIO::Read(is, payload);
137 }
138 return(rc);
139 }
140
141 //------------------------------------------------------------------------
145 //------------------------------------------------------------------------
146 template <typename T>
147 bool SetPayload(const T & payload)
148 {
149 bool rc = false;
150 Free();
151
152 uint16_t maxPayloadLen = 65535 - (20 + 8);
153 uint64_t payloadLen = IOUtils::StreamedLength(payload);
154 if (payloadLen <= maxPayloadLen) {
155 std::ostringstream os;
156 if (StreamIO::Write(os, payload)) {
157 _payload.first = payloadLen;
158 _payload.second = (uint8_t *)calloc(1, _payload.first);
159 assert(_payload.second);
160 memcpy(_payload.second, os.str().c_str(), _payload.first);
161 rc = true;
162 }
163 }
164 return(rc);
165 }
