181

266 bool NewFindLongest(const Ipv4Address & ipAddr,
267 std::pair<Ipv4Prefix,_valueT> & match) const
268 {
269 using std::map, std::pair, std::thread;
270 bool rc = false;
271 map<uint8_t,pair<bool,_valueT>> results;
272 std::map<uint8_t,thread> threads;
273 for (int i = 32; i >= 0; --i) {
274 if (! _hashMaps[i].empty()) {
275 results[i] = {false,_valueT()};
276 Ipv4Prefix pfx(ipAddr, i);
277 threads[i] = thread(&Ipv4Routes::FindInSubMap, this,
278 std::ref(_hashMaps[i]), pfx.Network(),
279 std::ref(results[i]));
280 }
281 }
282 for (auto it = threads.rbegin(); it != threads.rend(); ++it) {
283 it->second.join();
284 if (results[it->first].first) {
285 rc = true;
286 match.first = Ipv4Prefix(ipAddr, it->first);
287 match.second = results[it->first].second;
288 }
289 }
290 return rc;
291 }

298 bool FindLongest(const Ipv4Address & ipAddr,
299 std::pair<Ipv4Prefix,_valueT> & match) const
300 {
301 bool rc = false;
302
303 Ipv4Prefix lp(ipAddr, 32);
304 typename _RepSubType::const_iterator iter;
305 for (int8_t i = 32; i >= 0; --i) {
306 if (_hashMaps[i].empty())
307 continue;
308 lp.MaskLength(i);
309 iter = _hashMaps[i].find(lp.Network());
310 if (iter != _hashMaps[i].end()) {
311 match.first = lp;
312 match.second = iter->second;
313 rc = true;
314 break;
315 }
316 }
317 return(rc);
318 }

370 matches.push_back(match);

543 template <typename BinaryPredicate>
544 void Coalesce(BinaryPredicate pred)
545 {
546 // Combine adjacents.
547 for (uint8_t maskLen = 32; maskLen > 0; --maskLen) {
548 auto & hm = this->_hashMaps[maskLen];
549 auto hmi = hm.begin();
550 while (hmi != hm.end()) {
551 bool erased = false;
552 Ipv4Prefix pfx(hmi->first, maskLen);
553 Ipv4Prefix nextpfx = pfx;
554 ++nextpfx;
555 std::pair<bool,Ipv4Prefix> combpfx = pfx.Combine(nextpfx);
556 if (combpfx.first) {
557 auto nextit = hm.find(nextpfx.Network());
558 if (nextit != hm.end()) {
559 if (pred(hmi->second, nextit->second)) {
560 hm.erase(nextit);
561 _hashMaps[maskLen-1][combpfx.second.Network()] = hmi->second;
562 hmi = hm.erase(hmi);
563 erased = true;
564 }
565 }
566 }
567 if (! erased) {
568 ++hmi;
569 }
570 }
571 }
572 // Remove entries that are covered by matching entry with wider
573 // netmask.
574 for (uint8_t maskLen = 32; maskLen > 0; --maskLen) {
575 auto & hm = this->_hashMaps[maskLen];
576 auto hmi = hm.begin();
577 while (hmi != hm.end()) {
578 if (HaveWiderMatch(hmi->first, maskLen, hmi->second, pred)) {
579 hmi = hm.erase(hmi);
580 }
581 else {
582 ++hmi;
583 }
584 }
585 }
586 return;
587 }

618 void SortByKey(std::vector<std::pair<Ipv4Prefix,_valueT>> & target,
619 bool ascending = true) const
620 {
621 if (! target.empty())
622 target.clear();
623 if (! this->_hashMaps.empty()) {
624 target.resize(this->Size());
625 auto iter = this->_hashMaps.begin();
626 uint32_t pfx = 0;
627 for (uint8_t hashNum = 0; hashNum < 33; ++hashNum) {
628 if (! this->_hashMaps[hashNum].empty()) {
629 auto hiter = this->_hashMaps[hashNum].begin();
630 for ( ; hiter != iter->end(); ++hiter) {
631 target[pfx].first = Ipv4Prefix(hiter->first, hashNum);
632 target[pfx].second = hiter->second;
633 ++pfx;
634 }
635 }
636 }
637 if (! target.empty()) {
638 if (ascending) {
639 std::sort(target.begin(), target.end(), KeyLess());
640 }
641 else {
642 std::sort(target.begin(), target.end(), KeyGreater());
643 }
644 }
645 }
646
647 return;
648 }

688 uint32_t AddressesCovered() const
689 {
690 uint32_t rc = 0;
691 for (int8_t hashNum = 32; hashNum > 1; --hashNum) {
692 typename _RepSubType::const_iterator hiter =
693 _hashMaps[hashNum].begin();
694 for ( ; hiter != _hashMaps[hashNum].end(); ++hiter) {
695 bool foundWider = false;
696 for (int8_t widerHash = hashNum - 1; widerHash > 0; --widerHash) {
697 Ipv4Prefix widerPfx(hiter->first, widerHash);
698 if (_hashMaps[widerHash].find(widerPfx.Network())
699 != _hashMaps[widerHash].end()) {
700 // found wider match, don't count
701 foundWider = true;
702 break;
703 }
704 }
705 if (! foundWider) {
706 rc += ((uint32_t)1 << (32 - hashNum));
707 }
708 }
709 }
710 return rc;
711 }

756 template <typename BinaryPredicate>
757 bool HaveWiderMatch(const Ipv4Address & addr, uint8_t maskLen,
758 const _valueT & val, BinaryPredicate pred) const
759 {
760 bool rc = false;
761 if (maskLen > 0) {
762 for (int8_t wml = maskLen - 1; wml > 0; --wml) {
763 Ipv4Prefix pfx(addr, wml);
764 auto iter = _hashMaps[wml].find(pfx.Network());
765 if (iter != _hashMaps[wml].end()) {
766 if (pred(iter->second, val)) {
767 rc = true;
768 break;
769 }
770 }
771 }
772 }
773 return rc;
774 }
